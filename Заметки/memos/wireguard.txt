● Для доступа с NixOS на VPS через туннель нужно:

  # Проверить, есть ли маршрут до туннельной сети

  ping 10.100.0.2

  # Скопировать публичный ключ NixOS на VPS (через туннель)

  task deploy:wireguard-extract-vps-key
  ssh -i /tmp/vps-ssh-key.pem ubuntu@10.100.0.2 \
    "echo '$(cat ~/.ssh/id_ed25519.pub)' >> ~/.ssh/authorized_keys"

  # ~/.ssh/config
  Host vps
      HostName 10.100.0.2

  4. MikroTik продолжает слать keepalive каждые 25 сек
  5. Handshake завершается → туннель работает
